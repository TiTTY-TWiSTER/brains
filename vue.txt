	<li>{{item}}</li> - выводим элементы из массива

methods:{ # здесь создаем функции

@click - событие клик. Использование @click="название функции(можно без скобок,если не передаем параметры)"
@mouseover
@keyup.enter="передаем функцию" - при нажатии на клавишу(в данном случае enter) выполняем что-либо нами написанное

$event.target.value - получение значения из input

v-on:click="show = !show" - переключатель show = true или false
	Пример:
	<div id="demo">
  		<button v-on:click="show = !show">
   			 Переключить
  		</button>

  		<transition длительность :duration="{ enter: 500, leave: 800 }" name="fadeIn"> # анимация из библиотеки Animate.css
    			<p v-if="show">привет</p>
  		</transition>
	</div>
	так же в одном элементе несколько видов анимации:

	enter-active-class="animated fadeIn" - любая анимация появления
	leave-active-class="animated bounceOutRight" - исчезновения
