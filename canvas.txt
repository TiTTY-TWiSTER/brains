создаем тэг <canvas id = 'canvas'></canvas>

инициализируем его в js:
<script>
  var canv = document.getElementById('canvas');
  var ctx = canv.getContext('2d');

	задаем окно для рисования 
	конкретно здесь равное окну браузера

  canv.width = window.innerWidth;
  canv.height = window.innerHeight;

	рисуем через переменную ctx
 Квадрат!

  ctx.fillStyle = 'lightgreen' // цвет либо в HEX либо словами
  ctx.fillRect(50,50,300,200); // квадрат закрашенный (x,y,ширина,высота)
	
	ctx.strokeStyle = 'lightgreen'
	ctx.lineWidth = 10 // ширина линии
	ctx.strokeRect(50,50,300,200) // квадрат типа border 

 Круг!
 ctx.arc(canv.width / 2, canv.height / 2, radius, start angle, end angle,не обязательный параметр по часовой-false или против-true)
	
 Все что угодно !
	вызываем метод beginPath();
  ctx.beginPath();
  ctx.moveTo(50,50); начальная точка движения
  ctx.lineTo(25,100); прокладываем линию в другую позицию
  ctx.lineTo(75,100);
  ctx.lineTo(50,50);возвращаем в начальную точку и будет треугольник либо ctx.closePath();
  ctx.stroke();
  
 Текст!
	ctx.font = '30px Arial';
	ctx.fillText ('Hello World' , 20, 50);

 Градиент!
	var gradient = ctx.createLinearGradient(0,0,200,100);
	gradient.addColorStop(0,'blue'); // первый цвет
	gradient.addColorStop(1,'lightgreen'); // второй цвет
	ctx.fillStyle = gradient; // задаем цвет объекту кенваса
        ctx.fillRect(50,50,300,200); // квадрат с градиентом

	
  

</script>
